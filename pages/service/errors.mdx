import { useData } from 'nextra/data'


export const ErrorMapping = () => {
  const { mapping } = useData()
  return Object.entries(mapping).map(([key, value]) => <li>[<b>{key}</b>] - {value}</li>)
}

export const getStaticProps = async () => {
  const errors = await fetch("https://api.testnet.onbeam.com/v1/documentation/error-codes", {
    method: 'GET',
    headers: {
      "Content-Type": "application/json",
    },
  });

  return {
      props: {
        ssg: {
          mapping: errors
        }
      },
  }
}
