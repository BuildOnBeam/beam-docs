import { useData } from 'nextra/data'
import { errors } from '@onbeam/errors'


export const ErrorMapping = () => {
  const { mapping } = useData()
  return Object.entries(mapping).map(([key, value]) => <li>[<a href={`/service/errors/${key}`}>{key}</a>] - {value}</li>)
}
