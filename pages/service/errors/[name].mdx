import { useData } from 'nextra/data'
import { errors } from '@onbeam/errors'

export const ErrorName = () => {
  const { error } = useData()
  return error
}

export const ErrorDescription = () => {
  const { description } = useData()
  return description
}

# <ErrorName />

<ErrorDescription/>

export const getStaticProps = ({ params }) => {
  return {
      props: {
        ssg: {
          error: params.name,
          description: errors[params.name]
        }
      },
  }
}


export const getStaticPaths = (async () => {
  return {
    paths: [
      {
        params: {
          name: 'generic',
        },
      },
    ],
    fallback: true,
  }
})
